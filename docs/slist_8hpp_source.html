270template <typename T, typename A>
271void slist<T, A>::swap(slist &other) noexcept {
272 assert(node_alloc_ == other.node_alloc_);
273 node_base *new_tail = other.empty() ? &head_ : other.ptail_;
274 node_base *new_other_tail = empty() ? &other.head_ : ptail_;
275 swap(head_.next_, other.head_.next_);
276 swap(size_, other.size_);
277 ptail_ = new_tail;
278 other.ptail_ = new_other_tail;
279}

282template <typename... Args>
283typename slist<T, A>::iterator slist<T, A>::emplace(iterator it, Args &&... args) {
284 // TRACE(__PRETTY_FUNCTION__);
285 // node *new_node = node_alloc_.allocate(); // static_cast<node *>(...)
286 node *new_node = node_alloc_traits::allocate(node_alloc_, 1);
287 try {
288/*
289 Allocator-aware containers always call std::allocator_traits<A>::construct(m, p, args)
290 to construct an object of type T at p using args, with m == get_allocator().
291 The default construct in std::allocator calls ::new((void*)p) T(args) (until C++20)
292 std::allocator has no construct member and std::construct_at(p, args) is called when constructing elements (since C++20),
293 but specialized allocators may choose a different definition
294*/
295 // node_alloc_traits::construct(node_alloc_, addressof(new_node->value_), forward<Args>(args)...);
296 std::construct_at(addressof(new_node->value_), forward<Args>(args)...);
297 } catch (...) {
298 // Recover resources if exception on constructor call.
299 // node_alloc_.deallocate(new_node);
300 node_alloc_traits::deallocate(node_alloc_, new_node, 1);
301 // node_alloc_.deallocate(new_node, 1);
302 throw;
303 }
304 new_node->next_ = it.prev_->next_;
305 it.prev_->next_ = new_node;
306 if (it.prev_ == ptail_)
307 ptail_ = new_node; // Added at end
308 ++size_;
309 return it;
310}

313typename slist<T, A>::iterator slist<T, A>::erase(iterator b, iterator e) {
314 // TRACE(__PRETTY_FUNCTION__);
315 node *erase_next = b.prev_->next_;
316 node *erase_past = e.prev_->next_; // one past last erasure
317 if (nullptr == erase_past)
318 ptail_ = b.prev_; // Erasing at tail
319 b.prev_->next_ = erase_past; // splice out sublist
320 while (erase_next != erase_past) {
321 node *old_node = erase_next;
322 erase_next = erase_next->next_;
323 --size_;
324 // node_alloc_traits::destroy(node_alloc_, addressof(old_node->value_));
325 if constexpr (not std::is_fundamental_v<T>)

327 addressof(old_node->value_)->~T();
328 }
329 node_alloc_traits::deallocate(node_alloc_, old_node, 1);
330 // node_alloc_.deallocate(old_node, 1);
331 }
